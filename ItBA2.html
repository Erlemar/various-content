    frq = train[column].value_counts()/train.shape[0]
    ctc = frq.loc[frq.values<0.05].index
    
    for c in ctc:
        train[column].replace({c: 'Others'}, inplace = True)
        test[column].replace({c: 'Others'}, inplace = True)
